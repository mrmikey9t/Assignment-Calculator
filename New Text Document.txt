#include<iostream>
#include <cmath>
using namespace std;
void sum(int a,int b){
cout<<a+b<<endl;}

void sub(int a,int b){
cout<<a-b<<endl;}

void prod(int a,int b){
cout<<a*b<<endl;}

void div(float a,float b){
cout<<a/b<<endl;}

int main()
{
	int num1,num2;
	
cout<<"enter num1 = ";
cin>>num1;
cout<<"enter num2 = ";
cin>>num2;

cout<<"sum = ";
sum (num1,num2);
cout<<"subtract = ";
sub (num1,num2);
cout<<"product = ";
prod (num1,num2);
cout<<"division  = ";
div (num1,num2);
cout<<"log of num1 = ";
cout<<log(num1)<<endl;
cout<<"log of num2 = ";
cout<<log(num2)<<endl;
cout<<"square root of num1 = ";
cout<<sqrt(num1)<<endl;
cout<<"square root of num2 = ";
cout<<sqrt(num2)<<endl;
}
